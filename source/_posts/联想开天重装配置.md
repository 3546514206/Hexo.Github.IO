```shell
#!/bin/bash

# =============================================================================
# 信创笔记本重装后初始化脚本
# 功能：
#   1. 安装并启用 SSH
#   2. 将 nvme0n1p6 (294G) 挂载到 /home
#   3. 自动创建用户目录（如 setsunayang）
# 作者：Qwen
# 适用：Kylin OS / Ubuntu 系信创系统
# =============================================================================

set -e  # 遇错即停

LOG="/var/log/post-install.log"
exec > >(tee -a "$LOG") 2>&1

echo "=== 信创笔记本初始化脚本启动 === $(date)"

# ========================
# 1. 安装并配置 SSH
# ========================
echo "【1/3】安装 SSH 服务..."
apt update -y
apt install -y openssh-server

# 启用 SSH 开机自启
systemctl enable ssh
systemctl start ssh

# 允许 root 登录（按需，建议仅调试时开启）
# sed -i 's/#PermitRootLogin prohibit-password/PermitRootLogin yes/' /etc/ssh/sshd_config
# systemctl restart ssh

echo "✅ SSH 已安装并启动，可通过 ssh 用户名@$(hostname -I | awk '{print $1}') 连接"

# ========================
# 2. 准备 /home 分区
# ========================
echo "【2/3】配置 /home 使用大容量分区..."

# 卸载可能已自动挂载的 KTData
if mountpoint -q /media/"$USER"/KTData 2>/dev/null; then
    echo "卸载自动挂载的 KTData..."
    umount "/media/$USER/KTData"
fi

# 确保 /home 目录干净
if [ -d /home ] && [ "$(ls -A /home)" ]; then
    echo "备份现有 /home 到 /home_backup_$(date +%Y%m%d)..."
    cp -a /home /home_backup_$(date +%m%d)
    rm -rf /home/*
fi

# 获取 KTData 分区的 UUID（你的系统中固定为 c80ea973-...）
UUID_KTDATA="c80ea973-65a5-45b8-8aef-fc6a74fb8755"

# 检查分区是否存在
if ! blkid | grep -q "$UUID_KTDATA"; then
    echo "❌ 未找到 KTData 分区 (UUID: $UUID_KTDATA)，请检查磁盘！"
    exit 1
fi

# 挂载到 /home
mount "UUID=$UUID_KTDATA" /home

# 设置开机自动挂载（避免重复添加）
if ! grep -q "$UUID_KTDATA.*\/home" /etc/fstab; then
    echo "添加 /etc/fstab 条目..."
    echo "UUID=$UUID_KTDATA /home ext4 defaults 0 2" >> /etc/fstab
else
    echo "fstab 已配置，跳过。"
fi

echo "✅ /home 已挂载到 294G 分区"

# ========================
# 3. 创建默认用户目录
# ========================
echo "【3/3】创建用户家目录..."

# 假设你的用户名是 setsunayang（可修改）
DEFAULT_USER="setsunayang"

# 如果当前不是该用户，创建它
if ! id "$DEFAULT_USER" &>/dev/null; then
    echo "创建用户 $DEFAULT_USER..."
    useradd -m -s /bin/bash "$DEFAULT_USER"
    # 设置密码（可选，或留空让用户首次登录设置）
    # echo "$DEFAULT_USER:your_password" | chpasswd
fi

# 确保家目录存在且权限正确
HOME_DIR="/home/$DEFAULT_USER"
mkdir -p "$HOME_DIR"
chown "$DEFAULT_USER":"$DEFAULT_USER" "$HOME_DIR"

echo "✅ 用户 $DEFAULT_USER 的家目录已准备就绪"

# ========================
# 完成
# ========================
echo ""
echo "🎉 初始化完成！"
echo "   - SSH 已启用"
echo "   - /home 使用 294G 大分区"
echo "   - 用户 $DEFAULT_USER 可直接使用"
echo ""
echo "日志保存至: $LOG"
```
